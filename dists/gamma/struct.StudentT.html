The Student t distribution, t(nu), where nu is the degrees of freedom.

Example

extern crate dists;
extern crate rand;
use rand::Rng;
use dists::StudentT;

let t = StudentT::new(11.0);
let v: f64 = rand::thread_rng().gen(t);
println!("{} is from a t(11) distribution", v)
